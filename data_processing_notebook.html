# simple algorithm based on autocorrelation to calculate heart rate
# if you want to develop or run your own algorithm e-mail graham.healy@dcu.ie to request the raw data for this!
def calc_window_HR(x, nlags = 100, lowest_rate=40, hr_sps=100, hr_per_unit_time_seconds=60):
    # TODO: add in a HP filter and some other things for data cleaning... but works okay as is.. see plot
    acf_hr = statsmodels.tsa.stattools.acf(x, nlags=nlags )
    hr_ind = numpy.argmax(acf_hr[lowest_rate:])
    hr = hr_per_unit_time_seconds / ( (hr_ind + lowest_rate) / hr_sps ) 
    return hr

def ft_set_1(a, sensor_type=''):
    fts = collections.OrderedDict()
    if len(a)!=0:
        fts[sensor_type + '_median'] = numpy.median(a)
        fts[sensor_type + '_min'] = numpy.min(a)
        fts[sensor_type + '_max'] = numpy.max(a)
        fts[sensor_type + '_average'] = numpy.average(a)
        fts[sensor_type + '_std'] = numpy.std(a)
        #fts['mode'] = scipy.stats.mode(a)
        fts[sensor_type + '_len'] = len(a)
    else:
        fts[sensor_type + '_median'] = None
        fts[sensor_type + '_min'] = None
        fts[sensor_type + '_max'] = None
        fts[sensor_type + '_average'] = None
        fts[sensor_type + '_std'] = None
        #fts['mode'] = None
        fts[sensor_type + '_len'] = 0
        
    return fts
